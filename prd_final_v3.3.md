## 4. 🧠 학습심리학자 검토 (NEW)

### 적응형 학습 시스템 설계 원칙

```
┌─────────────────────────────────────────────────────────────────────┐
│                    학습과학 기반 설계 원칙                            │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  [1. 간격 반복 (Spaced Repetition)]                                 │
│  • 에빙하우스 망각곡선 기반 복습 주기 설정                           │
│  • 정답 시 간격 늘림, 오답 시 간격 줄임                              │
│  • 최적 타이밍에 자동 알림                                           │
│                                                                     │
│  [2. 능동적 회상 (Active Recall)]                                   │
│  • 단순 읽기가 아닌 퀴즈 기반 학습                                   │
│  • "생각해보기" 후 정답 확인                                         │
│  • 서술형 답변으로 깊은 이해 유도                                    │
│                                                                     │
│  [3. 교차 연습 (Interleaving)]                                      │
│  • 같은 주제만 반복 X, 다양한 주제 섞어서                            │
│  • 실제 면접처럼 예측 불가능한 순서                                  │
│                                                                     │
│  [4. 메타인지 (Metacognition)]                                      │
│  • "이 개념을 얼마나 이해했다고 생각하세요?" 자기 평가               │
│  • 실제 퀴즈 결과와 비교 → 자신감 보정                               │
│                                                                     │
│  [5. 성장 마인드셋 (Growth Mindset)]                                │
│  • "틀렸다" 대신 "아직 배우는 중"                                    │
│  • 오답을 성장 기회로 프레이밍                                       │
│  • 작은 성취도 인정하는 뱃지 시스템                                  │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

CREATE TABLE user_goals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    
    -- 목표 정보
    goal_type VARCHAR(50) NOT NULL,        -- job_interview, skill_up, certification
    target_company VARCHAR(100),           -- 카카오, 네이버, 삼성 등
    target_position VARCHAR(100),          -- 백엔드, 프론트엔드, 풀스택
    target_date DATE,                       -- 목표 달성 예정일
    
    -- 초기 상태
    initial_level_test_id UUID,
    background VARCHAR(50),                 -- cs_major, bootcamp, self_taught
    available_hours_per_week INTEGER,       -- 주당 가용 학습 시간
    
    status VARCHAR(20) DEFAULT 'active',   -- active, completed, paused
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- =====================================================
-- 맞춤 커리큘럼 테이블 (NEW)
-- =====================================================
CREATE TABLE curricula (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    goal_id UUID REFERENCES user_goals(id) ON DELETE CASCADE,
    
    -- 커리큘럼 정보
    title VARCHAR(200) NOT NULL,           -- "카카오 면접 준비 12주 커리큘럼"
    total_weeks INTEGER NOT NULL,
    current_week INTEGER DEFAULT 1,
    
    -- 진행 상태
    progress_percentage INTEGER DEFAULT 0,
    status VARCHAR(20) DEFAULT 'active',   -- active, completed, paused
    
    -- 커리큘럼 구조 (JSON)
    weekly_plan JSONB NOT NULL,            -- 주차별 학습 계획
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- =====================================================
-- 일일 학습 과제 테이블 (NEW)
-- =====================================================
CREATE TABLE daily_tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    curriculum_id UUID REFERENCES curricula(id) ON DELETE CASCADE,
    
    -- 과제 정보
    task_date DATE NOT NULL,
    task_type VARCHAR(50) NOT NULL,        -- new_concept, quiz, review, interview_sim
    category VARCHAR(50) NOT NULL,
    topic VARCHAR(100) NOT NULL,
    
    -- 상태
    status VARCHAR(20) DEFAULT 'pending',  -- pending, in_progress, completed, skipped
    completed_at TIMESTAMP,
    
    -- 메타
    estimated_minutes INTEGER,
    actual_minutes INTEGER,
    
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_daily_tasks_user_date ON daily_tasks(user_id, task_date);

-- =====================================================
-- 복습 스케줄 테이블 (Spaced Repetition) (NEW)
-- =====================================================
CREATE TABLE review_schedule (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    
    -- 복습 대상
    category VARCHAR(50) NOT NULL,
    topic VARCHAR(100) NOT NULL,
    concept_id VARCHAR(100),               -- 특정 개념 ID
    
    -- 간격 반복 데이터
    ease_factor DECIMAL(3,2) DEFAULT 2.5,  -- 난이도 계수
    interval_days INTEGER DEFAULT 1,        -- 현재 복습 간격
    repetition_count INTEGER DEFAULT 0,     -- 복습 횟수
    
    -- 다음 복습
    next_review_date DATE NOT NULL,
    last_review_date DATE,
    last_review_result VARCHAR(20),        -- easy, medium, hard, forgot
    
    -- 예측
    predicted_retention DECIMAL(3,2),       -- 예측 기억 유지율
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_review_schedule_next ON review_schedule(user_id, next_review_date);

-- =====================================================
-- 성취/업적 테이블 (NEW)
-- =====================================================
CREATE TABLE achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    
    -- 업적 정보
    achievement_type VARCHAR(50) NOT NULL,  -- badge, level_up, streak
    achievement_code VARCHAR(100) NOT NULL, -- streak_7_days, master_ds, etc.
    
    -- 상세
    title VARCHAR(200) NOT NULL,
    description TEXT,
    icon VARCHAR(50),
    
    -- XP
    xp_earned INTEGER DEFAULT 0,
    
    earned_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_achievements_user ON achievements(user_id);

-- =====================================================
-- 학습 스트릭 테이블 (NEW)

CREATE TABLE learning_streaks (

    -- 스트릭 정보

    longest_streak INTEGER DEFAULT 0,
    
    -- 보호권
    freeze_count INTEGER DEFAULT 0,         -- 남은 스트릭 보호권
    last_freeze_used DATE,
    
    -- 기록
    streak_start_date DATE,
    last_activity_date DATE,
    
    -- XP
    total_xp INTEGER DEFAULT 0,
    current_level INTEGER DEFAULT 1,

CREATE UNIQUE INDEX idx_learning_streaks_user ON learning_streaks(user_id);
